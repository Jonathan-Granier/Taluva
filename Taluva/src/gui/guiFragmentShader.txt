#version 130

in vec2 textureCoords;

out vec4 out_Color;

uniform sampler2D guiTexture;
uniform bool hover;
uniform bool clicked;
uniform bool grey;

void main(void){


	out_Color = texture(guiTexture,textureCoords);
	if(hover){
		if(clicked)
			out_Color.b +=1.0;
	}
	if(grey)
		out_Color = mix(out_Color,vec4(vec3(1/3,1/3,1/3),1.0),0.3);

}
